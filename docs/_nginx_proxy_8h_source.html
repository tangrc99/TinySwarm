53 std::string insertAddressPool(const std::string &pool_name, const AddressPool &pool) override {

118 void updateNginxStatus() {
119
120 std::string conf_str = basic_conf;
121
122 for (const auto &pool_: pools_) {
123 auto &name = pool_.first;
124 auto &pool = pool_.second.first;
125 auto &bind_port = pool_.second.second;
126 conf_str.insert(conf_str.size() - 1, toNginxConf(name, pool, bind_port));
127 }

129 writeConfFile(conf_str);

131 std::string reload = nginx_exec_.string() + " -s reload";

133 system(reload.c_str());

141 [[nodiscard]] static std::string
142 toNginxConf(const std::string &name, const AddressPool &pool, int bind_port) {
143 std::string result;
144 result.reserve(1000);
145
146
147 result += "\tupstream stream_" + name + " {\n";
148 for (auto &address: pool)
149 result += "\t\tserver " + address + " weight=1;\n";
150 result += "\t}\n";
151 result += "\tserver {\n"
152 "\t\tlisten " + std::to_string(bind_port) + ";\n"
153 "\t\tproxy_pass stream_" + name + ";\n"
154 "\t}\n";
155
156 return result;
157 }
