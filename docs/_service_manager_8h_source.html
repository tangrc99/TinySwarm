63 Json createService(std::string token, int num, const std::string &service,
64 ServiceType type, int port, const std::vector<char *> &exe_params,
65 const std::vector<char *> &docker_params, int restart = 0) {
66
67 // 如果没有指定名称，则需要随机指定
68 if (token.empty())
69 token = generateToken(9);
70
71 PodGroup group;
72
73 // 创建服务所指定数量的pod，pod名称为 service_xx
74 for (int i = 0; i < num; i++) {
75
76 auto alias = token;
77 if (num > 1)
78 alias += "_" + std::to_string(i);
79
80 auto res = manager_->selectWorkerToCreatePod(service, alias, type, port, exe_params, docker_params,
81 restart);
82
83 if (!res.isFail()) {
84 group.emplace_back(res.podDescriptor());
85 }
86 }
87
88 if (group.empty())
89 return {};
90
91 std::string addr;
92
93 if (port > 0) {
94 // 创建其代理地址
95 addr = proxy_->insertAddressPool(token, getPodGroupAddressPool(group));
96 }
97
98 ServiceDescriptor sd(token, group, num, addr, manager_.get());
99
100 auto res = map_.emplace(token, sd).first;
101
102 manager::ServiceDescriptor *sd_ptr = &res->second;
103
104 if (group.size() < num)
105 start_list_.emplace_back(sd_ptr);
106
107 return sd.toJson();
108 }

113 bool deleteService(const std::string &token) {
114
115 auto sd = getServiceDescriptor(token);
116
117 if (sd == nullptr)
118 return false;
119
120 sd->stopPodsInService();
121
122 if (sd->status != ServiceStatus::close) {
123 end_list_.emplace_back(sd);
124 } else {
125 removeService(sd);
126 }
127
128 // 无论关闭的状态是什么，都要关闭在代理处的注册
129 return proxy_->deleteAddressPool(token);
130 }
