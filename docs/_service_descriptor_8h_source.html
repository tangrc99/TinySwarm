77 void stopPodsInService() {
78
79 for (auto pod = pods_.begin(); pod != pods_.end();) {
80 auto[res, error] = manager_->stopPod((*pod));
81 if (res) {
82 pod = pods_.erase(pod);
83 } else {
84 pod++;
85 }
86 }
87
88 status = pods_.empty() ? close : hung;
89 }

93 int restartFailedPods() {
94
95 if (pods_.size() == num_)
96 return 0;
97
98 size_t size = pods_.size();
99
100 for (int i = 0; i < num_ - pods_.size(); i++) {
101
102 auto alias = token_;
103
104 alias += "_" + std::to_string(pods_.size() - 1);
105
106 auto res = manager_->selectWorkerToCreatePod(service, alias, type_, port_,
107 exe_params_, docker_params_, restart_);
108
109 if (!res.isFail()) {
110 pods_.emplace_back(res.podDescriptor());
111 }
112 }
113
114 if (num_ == pods_.size())
115 status = ready;
116
117 return static_cast<int>(pods_.size() - size);
118 }
