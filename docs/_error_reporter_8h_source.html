55 std::tuple<bool, std::string> insertExitInformation(const std::list<PodExitInformation> &infos) try {

57 std::vector<bsoncxx::document::view> docs;

59 for (auto &info: infos) {
60 auto doc = bsoncxx::builder::basic::make_document(

62 kvp("pod", info.alias),

67 kvp("signal", info.signal),

98 while (!quit_) {

100 try {
