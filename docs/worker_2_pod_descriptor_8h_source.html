44 struct PodExitInformation {
45 std::string alias; // 服务名称
46 std::string manager; // 创建该进程的manger
47 std::string out_file; // 进程的输出文件
48 std::string error_text; // 这里是存储 worker 可以确定的错误
49 int exit_num = 0; // 进程退出码
50 int signal = 0; // 信号量
51 bool core_dumped = false;
52
53 PodDescriptor desp_;
54
55 PodExitInformation(std::string alias_, std::string manager_, std::string out_file_, int status,
56 const PodDescriptor& desp, std::string error = "") : alias(std::move(alias_)),
57 manager(std::move(manager_)),
58 out_file(std::move(out_file_)), desp_(desp),
59 error_text(std::move(error)) {
60
61 if (status != -1) {
62 // 这里解析由进程退出收集来的信息
63 if (WIFSIGNALED(status)) {
64 signal = status << 8 >> 8;
65 exit_num = -1; // -1 代表是由信号退出
66 } else if (WIFEXITED(status)) {
67 exit_num = status >> 8;
68 }
69 core_dumped = WCOREDUMP(status);
70 }
71 }

75 [[nodiscard]] std::string getErrorText() const {
76 if (!error_text.empty())
77 return error_text;
78
79 std::string core_dump = core_dumped ? " core dumped" : "";
80 return signal > 0 ? "Interrupt by Signal " + std::to_string(signal) + core_dump :
81 "Finished with exit code " + std::to_string(exit_num) + core_dump;
82 }

87 [[nodiscard]] std::string getOutFile(int line = 10) const {
88 std::ifstream is(out_file, std::ios::ate);
89 is.seekg(-2, is.cur);
90
91
92 for (int i = 0; i < line && is.tellg() >= 0; i++) {
93 while (is.peek() != is.widen('\n') && is.tellg() >= 0) {
94 is.seekg(-1, is.cur);
95 }
96 is.seekg(-1, is.cur);
97 }
98
99 is.seekg(2, is.cur);
100
101 return {(std::istreambuf_iterator<char>(is)),
102 std::istreambuf_iterator<char>()};
