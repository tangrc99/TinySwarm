35 std::string insertAddressPool(const std::string &pool_name, const AddressPool &pool) override {
36
37 auto server = new IpvsServer({updateAddress(), pool});
38 pools_.emplace(pool_name, server);
39
40 std::string command = "ipvsadm -A -t ";
41 command += server->vs_address_ + " -s rr";
42
43 if (system(command.c_str()) != 0) {
44 throw ProxyException("IPVS -A ERROR");
45 }
46
47 for (auto &rs: server->rs_address_) {
48 command = "ipvsadm -a -t ";
49 command += rs + "-m";
50 if (system(command.c_str()) != 0) {
51 throw ProxyException("IPVS -a ERROR");
52 }
53 }
54
55 // 加入 CLUSTER_IP
56 command = "ipset add CLUSTER_IP hash:"; // FIXME: 这里有一个 ,
57 command += server->vs_address_;
58 if (system(command.c_str()) != 0) {
59 throw ProxyException("IPSET add ERROR");
60 }
61 // 加入 NODE_PORT
62
63 return server->vs_address_;

131 std::string updateAddress() {
